```markdown
# Agent Tool Projection Policy

This document is the canonical translation contract between shared agent
intent and harness-specific projection files.

## Shared Capability Vocabulary

- `Read`
- `Write`
- `Edit`
- `Glob`
- `Grep`
- `Bash`
- `USER_GATE`

## Canonical Meaning of Shared `tools:`

- The shared `tools:` field in `profile-al-dev-shared/agents/*.md` declares
  canonical capability intent for the shared authored agent.
- Shared agent bodies must not require a capability that is absent from the
  shared `tools:` list.
- Generated projections may rename capabilities for harness compatibility, but
  they must not silently drop or invent them.

## Projection Rules

### Claude

- Emit Markdown agent files.
- Keep explicit `tools:` lists for auditability.
- Translate `USER_GATE` to `AskUserQuestion`.

### Copilot

- Emit Markdown custom-agent files.
- Normalize capabilities to Copilot aliases.
- Omit `tools:` only when the shared file uses the full supported capability
  set and no restriction is intended.

### Codex

- Emit TOML agent config files.
- Encode capability needs inside `developer_instructions` and other
  Codex-native fields rather than inventing a synthetic `tools` key.
- Preserve inheritance defaults unless the projection policy requires an
  explicit override.

## Failure Semantics

- Unsupported shared capability in any projection is a hard error.
- Missing mapping metadata is a hard error.
- Shared body/frontmatter mismatch is a hard validation failure.
- Generated files are derived artifacts and must never be hand-edited.
```

```python
#!/usr/bin/env python3
import argparse
import json
from pathlib import Path

import yaml


REPO_ROOT = Path(__file__).resolve().parent.parent
AGENTS_DIR = REPO_ROOT / "profile-al-dev-shared" / "agents"
POLICY_PATH = REPO_ROOT / "profile-al-dev-shared" / "knowledge" / "agent-tool-projection-policy.md"
OUTPUT_ROOT = REPO_ROOT / "profile-al-dev-shared" / "generated" / "agents"


def split_frontmatter(text: str) -> tuple[dict, str]:
    if not text.startswith("---\n"):
        raise ValueError("Missing YAML frontmatter")
    _, fm, body = text.split("---\n", 2)
    return yaml.safe_load(fm), body


def parse_policy_frontmatter(text: str) -> dict:
    frontmatter, _ = split_frontmatter(text)
    return frontmatter


def translate_tools(tools: list[str], harness: str, policy: dict) -> list[str]:
    tool_map = policy["projection_rules"][harness].get("tool_map", {})
    translated = []
    for tool in tools:
        if tool not in tool_map:
            raise ValueError(f"Unsupported capability for {harness}: {tool}")
        mapped = tool_map[tool]
        if mapped not in translated:
            translated.append(mapped)
    return translated


def load_agent(path: Path) -> dict:
    frontmatter, body = split_frontmatter(path.read_text(encoding="utf-8"))
    return {
        "name": path.stem,
        "description": frontmatter["description"].strip(),
        "model": frontmatter["model"],
        "tools": frontmatter["tools"],
        "body": body.lstrip(),
    }


def render_markdown_projection(agent: dict, tools: list[str]) -> str:
    tool_list = ", ".join(json.dumps(tool) for tool in tools)
    return (
        "---\n"
        f'name: {agent["name"]}\n'
        f'description: {agent["description"]}\n'
        f'model: {agent["model"]}\n'
        f"tools: [{tool_list}]\n"
        "---\n\n"
        f'{agent["body"]}'
    )


def render_codex_projection(agent: dict, policy: dict) -> str:
    instructions = (
        f'{agent["body"].strip()}\n\n'
        f"Shared capabilities: {', '.join(agent['tools'])}.\n"
        "Honor inherited sandbox and MCP configuration unless the parent session overrides them."
    )
    escaped = json.dumps(instructions)
    return (
        f'name = "{agent["name"]}"\n'
        f'description = {json.dumps(agent["description"])}\n'
        f"developer_instructions = {escaped}\n"
    )


def write_projection(path: Path, content: str) -> None:
    path.parent.mkdir(parents=True, exist_ok=True)
    path.write_text(content, encoding="utf-8")


def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--check", action="store_true")
    args = parser.parse_args()

    policy = parse_policy_frontmatter(POLICY_PATH.read_text(encoding="utf-8"))
    for agent_path in sorted(AGENTS_DIR.glob("*.md")):
        agent = load_agent(agent_path)
        claude_tools = translate_tools(agent["tools"], "claude", policy)
        copilot_tools = translate_tools(agent["tools"], "copilot", policy)

        write_projection(
            OUTPUT_ROOT / "claude" / agent_path.name,
            render_markdown_projection(agent, claude_tools),
        )
        write_projection(
            OUTPUT_ROOT / "copilot" / agent_path.name,
            render_markdown_projection(agent, copilot_tools),
        )
        write_projection(
            OUTPUT_ROOT / "codex" / f"{agent['name']}.toml",
            render_codex_projection(agent, policy),
        )


if __name__ == "__main__":
    main()
```

```python
def load_projection_policy(plugin_root: Path) -> dict:
    policy_path = plugin_root / "knowledge" / "agent-tool-projection-policy.md"
    if not policy_path.exists():
        raise FileNotFoundError(f"Projection policy not found at {policy_path}")
    text = policy_path.read_text(encoding="utf-8")
    if not text.startswith("---\n"):
        raise ValueError("Projection policy missing YAML frontmatter")
    _, fm, _ = text.split("---\n", 2)
    return json.loads(json.dumps(__import__("yaml").safe_load(fm)))


def validate_shared_tools(tools: list[str], allowed: set[str]) -> None:
    for tool in tools:
        if tool not in allowed:
            raise ValueError(f"Unknown shared capability: {tool}")


def detect_required_capabilities(body: str) -> set[str]:
    required = set()
    if "USER_GATE" in body:
        required.add("USER_GATE")
    if "shell" in body.lower() or "bash" in body.lower() or "al-compile" in body:
        required.add("Bash")
    return required


def validate_body_tool_consistency(tools: list[str], required: set[str]) -> None:
    missing = sorted(required - set(tools))
    if missing:
        raise ValueError(
            f"Body requires capabilities missing from tools: {', '.join(missing)}"
        )
```
